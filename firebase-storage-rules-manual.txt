// Copy and paste these rules into Firebase Console -> Storage -> Rules

rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    // Public read access to all files
    match /{allPaths=**} {
      allow read: if true;
    }
    
    // Allow photo uploads directly to trip-photos folder
    match /trip-photos/{filename} {
      allow write: if request.resource != null 
                   && request.resource.contentType.matches('image/.*') 
                   && request.resource.size < 10 * 1024 * 1024;
      allow delete: if true;
    }
    
    // Allow test uploads for diagnostics
    match /test-uploads/{filename} {
      allow write: if request.resource != null 
                   && request.resource.contentType.matches('image/.*') 
                   && request.resource.size < 10 * 1024 * 1024;
      allow delete: if true;
    }
  }
}

// Steps to apply these rules:
// 1. Go to https://console.firebase.google.com/project/bigmountainclubwebsite/storage/rules
// 2. Replace the existing rules with the rules above
// 3. Click "Publish"
